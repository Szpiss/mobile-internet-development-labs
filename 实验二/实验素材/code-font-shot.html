/**
 * 切换文章字号并同步更新状态。
 * @param {"small" | "medium" | "large"} size 目标字号类型。
 * @returns {void} 本函数仅更新页面展示状态。
 */
function changeFontSize(size) {
  const sizeMap = {
    small: { fontSize: "var(--small-font)", label: "小号字体" },
    medium: { fontSize: "var(--medium-font)", label: "中号字体" },
    large: { fontSize: "var(--large-font)", label: "大号字体" }
  };

  articleText.style.fontSize = sizeMap[size].fontSize;
  fontStatus.textContent = sizeMap[size].label;
}
