// JavaScript 中初始化 Swiper 竖向切换
var swiper = new Swiper(".invite-swiper", {
  direction: "vertical",
  loop: false,
  speed: 620,
  mousewheel: true,
  pagination: {
    el: ".swiper-pagination",
    clickable: true
  },
  on: {
    init: function () {
      runAnimations(this.slides[this.activeIndex]);
    },
    slideChangeTransitionEnd: function () {
      runAnimations(this.slides[this.activeIndex]);
    }
  }
});
